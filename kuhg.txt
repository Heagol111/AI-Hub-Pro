<script>
  /* THEME: explicit 'light' or 'dark' + localStorage */
  const themeBtn = document.getElementById('themeToggle');
  function setTheme(t){
    document.body.setAttribute('data-theme', t);
    const icon = document.getElementById('icon');
    if(t === 'dark'){ icon.classList.remove('fa-moon'); icon.classList.add('fa-sun'); }
    else { icon.classList.remove('fa-sun'); icon.classList.add('fa-moon'); }
    localStorage.setItem('aihub-theme', t);
  }
  (function initTheme(){
    const saved = localStorage.getItem('aihub-theme');
    if(saved) { setTheme(saved); return; }
    const preferDark = window.matchMedia && window.matchMedia('(prefers-color-scheme:dark)').matches;
    setTheme(preferDark ? 'dark' : 'light');
  })();
  themeBtn.addEventListener('click', () => {
    const cur = document.body.getAttribute('data-theme') === 'dark' ? 'dark' : 'light';
    setTheme(cur === 'dark' ? 'light' : 'dark');
  });

  /* SEARCH + TABS FILTER */
  const tabs = document.querySelectorAll('.tab');
  const cards = Array.from(document.querySelectorAll('.card'));
  const searchInput = document.getElementById('search');

  tabs.forEach(tab=>{
    tab.addEventListener('click', ()=>{
      tabs.forEach(t=>t.classList.remove('active'));
      tab.classList.add('active');
      applyFilter();
    });
  });

  searchInput.addEventListener('input', applyFilter);

  function applyFilter(){
    const q = searchInput.value.trim().toLowerCase();
    const activeCat = document.querySelector('.tab.active').dataset.cat;

    cards.forEach(card=>{
      const title = (card.dataset.title || '').toLowerCase();
      const desc = (card.querySelector('.card-desc')?.textContent || '').toLowerCase();
      const cat = card.dataset.cat || 'all';

      const matchesCat = (activeCat === 'all') || (cat === activeCat);
      const matchesQuery = !q || title.includes(q) || desc.includes(q);
      card.style.display = (matchesCat && matchesQuery) ? 'block' : 'none';
    });
  }

  // Accessibility: press '/' to focus search
  window.addEventListener('keydown', (e)=>{
    if(e.key === '/' && document.activeElement !== searchInput){
      e.preventDefault();
      searchInput.focus();
    }
  });

  // ensure initial filter state
  applyFilter();
</script>
